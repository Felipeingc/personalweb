    // Mover el jugador con el mouse

    document.addEventListener("mousemove", function (event) {

        const projectGame = document.getElementById("projectGame");

        if (projectGame && projectGame.style.display === "block") {

            const gameRect = projectGame.getBoundingClientRect();

            playerPosition = event.clientX - gameRect.left - playerWidth / 2;

            if (playerPosition < 0) playerPosition = 0;

            if (playerPosition > gameWidth - playerWidth) playerPosition = gameWidth - playerWidth;

            document.getElementById("player").style.left = playerPosition + "px";

        }

    });

    }

    // Función para mostrar u ocultar el contenido de cada tarjeta individualmente

    function toggleCard(element) {

      const content = element.nextElementSibling;

      if (content.classList.contains('show')) {

        content.classList.remove('show');

        element.classList.remove('active');

      } else {

        content.classList.add('show');

        element.classList.add('active');

      }

    }


    // Animación al hacer scroll para secciones y elementos

    const hiddenElements = document.querySelectorAll('.hidden');


    const observerOptions = {

      threshold: 0.1,

    };


    const observer = new IntersectionObserver((entries, observer) => {

      entries.forEach(entry => {

        if (entry.isIntersecting) {

          entry.target.classList.add('show');

          entry.target.classList.remove('hidden');

          observer.unobserve(entry.target);

        }

      });

    }, observerOptions);


    hiddenElements.forEach(element => {

      observer.observe(element);

    });


    // Animación para las tarjetas de servicio, proyectos y testimonios

    const services = document.querySelectorAll('.service-card');

    const projects = document.querySelectorAll('.project');

    const testimonials = document.querySelectorAll('.testimonial');


    services.forEach(service => {

      observer.observe(service);

    });


    projects.forEach(project => {

      observer.observe(project);

    });


    testimonials.forEach(testimonial => {

      observer.observe(testimonial);

    });
